import React from "react"
/**
 * SelectSpeaker
 * state = {speakerId}
 * props = {config:AppConfig}
 * @component
 * @example
 * const config = AppConfig.getInstance()
 * <SelectSpeaker config={config}/>
 * */
class SelectSpeaker extends React.Component{

	constructor(props){
		super(props)
		this.config = props.config
		this.state = {
			speakerId : localStorage.lastSpeakerId||'wibowo'
		}
	}

	formatSpeakerName(speaker){
		let speaker_display_name = speaker.alias != '' ? speaker.alias : speaker.name;
		return speaker_display_name = `${speaker_display_name}/${speaker.gender}/${speaker.age}/${speaker.fast?'Fast':'Low'} ` 
	}
	onChangeSpeakerId(evt){
		const speakerId = $(evt.target).find('option:selected').val();
		this.setState({speakerId});
		localStorage.lastSpeakerId = speakerId;
	}

	render(){
		const inputDefaultCls = "py-3 px-4 block w-full border-gray-200 rounded-md text-sm focus:border-blue-500 focus:ring-blue-500 dark:bg-slate-900 dark:border-gray-700 dark:text-gray-400";

		return(<div  style={{zIndex:15}} className="fixed  top-0 right-0 columns-2">
		<label className={inputDefaultCls}>Select Speaker</label><select id="speaker_id" className={inputDefaultCls} onChange={ evt => this.onChangeSpeakerId(evt)} 
				value={this.state.speakerId}>
			{
				speaker_ids.map((speaker,index)=>{
					return(<option key={index} value={speaker.name}>{this.formatSpeakerName(speaker)}</option>)
				})
			}            
      
    </select>
    </div>)
	}
}


export default SelectSpeaker
